CREATE VIEW [dbo].[APLICATIVOS_APROBADOS]
AS
SELECT        a.nombre_app, rolesA.rol_id, rolesA.app_id
FROM            dbo.APLICATIVOS AS a INNER JOIN
                             (SELECT        ra.rol_id, p.app_id, p.user_id, p.estado
                               FROM            dbo.PERMISOS AS p INNER JOIN
                                                         dbo.ROLES_NEGOCIO_APLICATIVOS AS ra ON p.app_id = ra.app_id AND p.rol_neg_id = ra.rol_neg_id) AS rolesA ON a.app_id = rolesA.app_id
WHERE        (rolesA.user_id = SESSION_CONTEXT(N'user_id')) AND (rolesA.estado = 'ACTIVO')
GROUP BY a.nombre_app, rolesA.rol_id, rolesA.app_id
